# ShadowDragon Filters
# Format: [\] filter [/] description [X] menutitlename

[X] ShadowDragon (Filters)

# -------------------------
# ENTITY TYPE FILTERS
# -------------------------

[\] type:person
[/] Show only individual profiles

[\] type:organization
[/] Show only companies or organizations

[\] type:username
[/] Show only usernames

[\] type:email
[/] Show only email addresses

[\] type:phone
[/] Show only phone numbers

[\] type:domain
[/] Show only domains

[\] type:ip
[/] Show only IP addresses

[\] type:crypto
[/] Show cryptocurrency wallets

[\] type:image
[/] Show image results

# -------------------------
# PLATFORM FILTERS
# -------------------------

[\] platform:twitter
[/] Show results from Twitter/X

[\] platform:facebook
[/] Show results from Facebook

[\] platform:instagram
[/] Show results from Instagram

[\] platform:linkedin
[/] Show results from LinkedIn

[\] platform:youtube
[/] Show results from YouTube

[\] platform:tiktok
[/] Show results from TikTok

[\] platform:reddit
[/] Show results from Reddit

[\] -platform:pinterest
[/] Exclude Pinterest results

# -------------------------
# ACCOUNT INFORMATION
# -------------------------

[\] verified:true
[/] Show verified accounts

[\] followers:>1000
[/] Accounts with more than 1000 followers

[\] followers:<100
[/] Accounts with fewer than 100 followers

[\] posts:>500
[/] Accounts with more than 500 posts

[\] activity:recent
[/] Recently active accounts
