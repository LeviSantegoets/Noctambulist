# OpenSSL Command List
# Format: [\] command [/] description [X] menutitlename

[X] OpenSSL Command List

[\] openssl version
[/] Show OpenSSL version

[\] openssl version -a
[/] Show full OpenSSL build info

[\] openssl s_client -connect example.com:443
[/] Connect to an HTTPS server and show certificate

[\] openssl s_client -connect example.com:443 -servername example.com
[/] Test SNI (Server Name Indication)

[\] openssl s_client -connect example.com:443 -showcerts
[/] Show full certificate chain

[\] openssl s_client -connect example.com:443 -tls1_2
[/] Force TLS 1.2

[\] openssl s_client -connect example.com:443 -tls1_3
[/] Force TLS 1.3

[\] openssl x509 -in cert.pem -text -noout
[/] Display certificate details

[\] openssl x509 -in cert.pem -issuer -noout
[/] Show certificate issuer

[\] openssl x509 -in cert.pem -subject -noout
[/] Show certificate subject

[\] openssl x509 -in cert.pem -enddate -noout
[/] Show certificate expiration date

[\] openssl rsa -in key.pem -check
[/] Check RSA private key

[\] openssl rsa -in key.pem -pubout > pubkey.pem
[/] Extract public key from private key

[\] openssl genrsa -out key.pem 2048
[/] Generate a 2048‑bit RSA private key

[\] openssl genpkey -algorithm RSA -out key.pem -pkeyopt rsa_keygen_bits:4096
[/] Generate a 4096‑bit RSA key

[\] openssl genpkey -algorithm ED25519 -out key.pem
[/] Generate an Ed25519 key

[\] openssl req -new -key key.pem -out request.csr
[/] Create a certificate signing request (CSR)

[\] openssl req -new -x509 -key key.pem -out cert.pem -days 365
[/] Create a self‑signed certificate

[\] openssl pkcs12 -export -out cert.pfx -inkey key.pem -in cert.pem
[/] Create a PFX/P12 certificate bundle

[\] openssl pkcs12 -in cert.pfx -nodes -out output.pem
[/] Extract PEM files from a PFX

[\] openssl enc -aes-256-cbc -salt -in file.txt -out file.enc
[/] Encrypt a file with AES‑256

[\] openssl enc -aes-256-cbc -d -in file.enc -out file.txt
[/] Decrypt a file

[\] openssl dgst -sha256 file.txt
[/] Generate SHA‑256 hash of a file

[\] openssl rand -hex 32
[/] Generate 32 random bytes (hex)

[\] openssl speed
[/] Benchmark crypto performance

[\] openssl speed aes-256-cbc
[/] Benchmark AES‑256

[\] openssl s_server -key key.pem -cert cert.pem -port 4443
[/] Start a simple TLS server

[\] openssl s_client -connect localhost:4443
[/] Connect to the OpenSSL TLS server
